(define (domain flat-tyre-typing)
	 (:requirements :strips :typing)
	 (:types container nut hub tool wheel)
	 (:constants
		wrench - tool
		pump - tool
		jack - tool)
	 (:predicates
		(locked ?c - container)
		(open ?c - container)
		(in ?x - (either tool wheel) ?c - container)
		(have ?x - (either tool nut wheel))
		(tight ?n - nut ?h - hub)
		(loose ?n - nut ?h - hub)
		(on-ground ?h - hub)
		(unfastened ?h - hub)
		(on ?n - (either nut wheel) ?h - hub)
		(free ?h - hub)
		(inflated ?w - wheel)
		(intact ?w - wheel))
	(:action open-container
		:parameters (?c - container))
	(:action close-container
		:parameters (?c - container))
	(:action fetch0
		:parameters (?x - tool ?c - container))
	(:action fetch1
		:parameters (?x - wheel ?c - container))
	(:action put-away0
		:parameters (?x - tool ?c - container))
	(:action put-away1
		:parameters (?x - wheel ?c - container))
	(:action loosen
		:parameters (?n - nut ?h - hub))
	(:action tighten
		:parameters (?n - nut ?h - hub))
	(:action jack-up
		:parameters (?h - hub))
	(:action jack-down
		:parameters (?h - hub))
	(:action remove-nuts
		:parameters (?n - nut ?h - hub))
	(:action put-on-nuts
		:parameters (?n - nut ?h - hub))
	(:action remove-wheel
		:parameters (?w - wheel ?h - hub))
	(:action put-on-wheel
		:parameters (?w - wheel ?h - hub))
	(:action inflate
		:parameters (?w - wheel))
)
